def camelcase
  gsub(/\w+/) {|chunk|
    ''.tap {|s|
      chunk.split(DELIMITER).each_with_index do |word, index|
        s << case index
        when 0
          word.downcase
        else
          word.extend(StringExtension).force_pascalcase
        end
      end
    }
  }
end
